#include <tunables/global>

  #include <abstractions/base>

  # Capabilities (restrict to what's actually needed)
  capability net_bind_service,
  capability setuid,
  capability setgid,
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # Network (Modbus TCP + MQTT only)
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # S6-Overlay init system
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /sbin/** ix,
  /usr/sbin/** ix,
  /lib/** ix,
  /usr/lib/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,

  # Standard devices
  /dev/tty rw,
  /dev/null rw,
  /dev/zero rw,
  /dev/random r,
  /dev/urandom r,

  # Bashio (Home Assistant helper library)
  /usr/lib/bashio/** ix,

  # Run script
  /run.sh rix,

  # Python runtime
  /usr/bin/python3* ix,

  # Application code (read-only is correct!)
  /app/** r,

  # Data directory (persistent storage)
  /data/** rw,

  # Temporary files
  /tmp/** rwk,
  /var/tmp/** rwk,

  # System files (read-only)
  /etc/** r,
  /proc/** r,
  /sys/class/** r,
  /sys/devices/** r,

  # Python standard library and site-packages

  /usr/local/lib/python3*/** r,

  # Supervisor communication (WICHTIG f√ºr Backups!)
  /run/supervisor.sock rw,
  owner /data/.backup.lock rwk,

  # Allow supervisor to snapshot our data
  /data/.snapshot/** rw,

  # Deny dangerous writes
  deny /proc/sys/kernel/** w,
  deny /sys/firmware/** w,
  deny /sys/kernel/** w,
  deny /proc/kcore r,
}
